//Cálculo del caudal medio anual.
var caudalAnual = ($feature.Octubre + $feature.Noviembre + $feature.Diciembre + $feature.Enero + $feature.Febrero + $feature.Marzo + $feature.Abril + $feature.Mayo + $feature.Junio + $feature.Julio + $feature.Agosto + $feature.Septiembre) / 12;

//Redondea la cifra del caudal a 2 decimales.
var caudalAnualFinal = round(caudalAnual, 2)

//Devuelve el nombre del municipio con la primera letra en mayúscula y (Sin datos), en caso de no haber datos sobre el caudal.
if (caudalAnualFinal <= 0){
Proper($feature.Municipio, 'everyword') + ' (Sin datos)';
}
//Devuelve el nombre del municipio con la primera letra en mayúscula y el cálculo del caudal redondeado, acompañado de su unidad de medida.
else {
Proper($feature.Municipio, 'everyword') + ' ' + caudalAnualFinal + ' m3/s';
};
